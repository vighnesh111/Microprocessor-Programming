;WAP to display the message “TE EXTC” on IBM PC. Use INT 21h function, AH=09 with string of message at DS: DX and terminated by “$”.
DATA SEGMENT
MSG1 DB 0AH,0DH, "TE EXTC $"
DATA ENDS
STACK SEGMENT
STACK ENDS
EXTRA SEGMENT
EXTRA ENDS
CODE SEGMENT
ASSUME CS: CODE, DS: DATA, ES: EXTRA, SS: STACK
START: MOV AX, DATA
MOV DS, AX
MOV AX, STACK
MOV SS, AX
MOV AX, EXTRA
MOV ES, AX
MOV AH, 09H
LEA DX, MSG1
INT 21H
CODE ENDS
END START
