;WAP to transfer a block of 10 bytes from data segment to extra segment
DATA SEGMENT
BLOCK1 DB 05H, 36H, 01H, 09H, 78H, 14H, 96H, 78H, 15H, 12H
DATA ENDS
EXTRA SEGMENT
BLOCK2 DB 10 DUP (?)
EXTRA ENDS
CODE SEGMENT
ASSUME CS: CODE, DS: DATA, ES: EXTRA
MOV AX, DATA
MOV DS, AX
MOV AX, EXTRA
MOV ES, AX
LEA SI, BLOCK1
LEA DI, BLOCK2
MOV CX, 00AH
CLD
REP MOVSB
CODE ENDS
END
