;WAP to multiply two 16 bit numbers. Operands and result in data segment
.MODEL SMALL
.STACK 100H
.DATA
MULTIPLIER DW 1234H
MULTIPLICAND DW 3456H
PRODUCT DD ?
.CODE
MULT PROC
MOV AX, @DATA
MOV DS, AX
MOV AX, MULTIPLICAND
MUL MULTIPLIER
MOV PRODUCT, AX
MOV PRODUCT+02H, DX
MOV AH, 04CH
INT 21H
MULT ENDP
END MULT
RET
