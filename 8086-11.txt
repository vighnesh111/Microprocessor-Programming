;WAP to sort a series of 10 numbers from 20000H in ascending order
CODE SEGMENT
ASSUME CS: CODE
MOV AX, 2000H
MOV DS, AX
MOV CH, 09H
BACK2:
MOV SI, 0000H
MOV CL, 09H
BACK1:
MOV AX, [SI]
CMP AH, AL
JNC SKIP
XCHG AL, AH
MOV [SI], AX
SKIP:
INC SI
DEC CL
JNZ BACK1
DEC CH
JNZ BACK2
INT 03H
CODE ENDS
END
