;WAP to perform addition of two 16 bit BCD numbers
DATA SEGMENT
A DW 1234H
B DW 5140H
SUM DW ?
CARRY DB 00H
DATA ENDS
CODE SEGMENT
ASSUME CS: CODE, DS: DATA
MOV AX, DATA
MOV DS, AX
MOV AX, A
MOV BX, B
ADD AL, BL
DAA
MOV AL, AH
ADC AL, BH
DAA
JNC SKIP
MOV CARRY, 01H
SKIP: MOV SUM, AX
INT 03H
CODE ENDS
END
