;WAP to convert BCD number to binary or hexadecimal number
MOV SI, 2000H
MOV AX, [SI]
MOV DX, 0000H
MOV BX, AX
AND BH, 0F0H
MOV CL, 04H
ROR BH, CL
CMP BH, 00H
JZ NEXT1
BACK1:
ADD DX, 03E8H
DEC BH
JNZ BACK1
NEXT1:
AND AH, 00FH
CMP AH, 00H
JZ NEXT2
BACK2:
ADD DX, 64H
DEC AH
JNZ BACK2
NEXT2:
AND BL, 0F0H
ROR BL, CL
CMP BL, 00H
JZ NEXT3
BACK3:
ADD DX, 00AH
DEC BL
JNZ BACK3
NEXT3:
AND AL, 00FH
ADD DX, AX
HLT
;(input in 2000H and 2001H)
